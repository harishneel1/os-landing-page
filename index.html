agent.js
// Initialize OpenSlate agent
import { Agent } from '@openslate/sdk';

const agent = new Agent({
  apiKey: process.env.OPENSLATE_API_KEY,
  model: 'gpt-4',
  temperature: 0.7
});

// Configure agent capabilities
await agent.addCapability({
  name: 'search',
  provider: 'neural-search'
});
